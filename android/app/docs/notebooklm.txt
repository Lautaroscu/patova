import re

def clean_number(phone_str):
    """
    Limpia el formato del número eliminando signos, espacios y el prefijo internacional
    para operar con enteros puros (ej: +54 9 11 5438-6201 -> 5491154386201)
    """
    cleaned = re.sub(r'\D', '', phone_str)
    return int(cleaned) if cleaned else None

def generate_block_range(seed_number, block_size=1000):
    """
    Toma un número semilla y calcula el inicio y fin del bloque de 1000 números.
    Ejemplo: 5491154386201 -> Inicio: 5491154386000, Fin: 5491154386999
    """
    base = (seed_number // block_size) * block_size
    range_start = base
    range_end = base + (block_size - 1)
    return range_start, range_end

def seed_spam_database(seed_list):
    conn = None
    try:
        conn = psycopg2.connect(**DB_PARAMS)
        cur = conn.cursor()
        
        # 1. Crear tabla de ejemplo si no existe
        cur.execute("""
            CREATE TABLE IF NOT EXISTS spam_numbers (
                phone_number BIGINT PRIMARY KEY,
                is_predicted BOOLEAN DEFAULT FALSE,
                detected_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            );
            CREATE INDEX IF NOT EXISTS idx_spam_numbers_phone ON spam_numbers(phone_number);
        """)
        
        all_numbers_to_insert = set()
        
        # 2. Procesar cada semilla para expandir el rango
        for seed_str in seed_list:
            cleaned = clean_number(seed_str)
            if not cleaned:
                continue
                
            start, end = generate_block_range(cleaned)
            print(f"Expandiendo semilla {seed_str} a rango: {start} - {end}")
            
            # Generamos los 1000 números del bloque
            for num in range(start, end + 1):
                # El número original se marca como detectado real, los otros como predicción
                is_predicted = (num != cleaned)
                all_numbers_to_insert.add((num, is_predicted))
        
        # Convertimos a lista para psycopg2
        data_tuples = list(all_numbers_to_insert)
        
        # 3. Inserción masiva eficiente ignorando duplicados existentes (Upsert)
        query = """
            INSERT INTO spam_numbers (phone_number, is_predicted)
            VALUES %s
            ON CONFLICT (phone_number) DO NOTHING;
        """
        
        print(f"Iniciando inserción masiva de {len(data_tuples)} números...")
        execute_values(cur, query, data_tuples)
        conn.commit()
        print("¡Base de datos seedeada con éxito!")
        
    except Exception as e:
        if conn:
            conn.rollback()
        print(f"Error durante el seeding: {e}")
    finally:
        if conn:
            cur.close()
            conn.close()

def clean_and_get_block_base(raw_phone: str, block_size: int = 1000):
    """Limpia el número y devuelve el identificador base del millar."""
    cleaned = re.sub(r'\D', '', raw_phone)
    if not cleaned:
        return None, None
    phone_int = int(cleaned)
    # División entera para truncar al millar más cercano (ej: 5491154386201 -> 5491154386000)
    block_base = (phone_int // block_size) * block_size
    return phone_int, block_base

def process_expansion(cur, unique_seeds_with_bases):
    """
    Genera los rangos de 1000 números para cada base única y los inserta en la BD.
    Garantiza que los reportes reales se guarden con is_predicted = FALSE y las predicciones con TRUE.
    """
    all_numbers_to_upsert = {}
    
    # Primero, calculamos las expansiones de 1000 números por cada bloque base único
    for seed, base in unique_seeds_with_bases.items():
        for num in range(base, base + 1000):
            # Por defecto lo marcamos como predicción (TRUE)
            all_numbers_to_upsert[num] = True

    # Segundo, sobreescribimos los números reales reportados como FALSE (reactivos)
    for seed in unique_seeds_with_bases.keys():
        all_numbers_to_upsert[seed] = False

    # Preparamos las tuplas para la inserción masiva
    data_tuples = [(num, is_pred) for num, is_predicted in all_numbers_to_upsert.items()]

    # Query optimizada con UPSERT. Si el número ya existía, actualiza 'is_predicted' 
    # (por si pasa de ser preventivo a ser un reporte real confirmado por el bot)
    query = """
        INSERT INTO spam_numbers (phone_number, is_predicted)
        VALUES %s
        ON CONFLICT (phone_number) 
        DO UPDATE SET is_predicted = EXCLUDED.is_predicted
        RETURNING phone_number;
    """
    
    execute_values(cur, query, data_tuples)
    inserted_count = len(cur.fetchall())
    return inserted_count

@app.post("/api/v1/reports/batch", status_code=status.HTTP_201_CREATED)
async def report_spam_batch(payload: AndroidReportBatch):
    """
    Endpoint utilizado por los clientes Android para sincronizar periódicamente
    los números bloqueados localmente. Dispara la expansión automática por rangos.
    """
    if not payload.numbers:
        raise HTTPException(status_code=400, detail="La lista de números no puede estar vacía.")
    
    unique_seeds_with_bases = {}
    
    # Agrupamos por bloques en memoria antes de tocar la base de datos
    for num_str in payload.numbers:
        phone_int, block_base = clean_and_get_block_base(num_str)
        if phone_int and block_base:
            unique_seeds_with_bases[phone_int] = block_base

    try:
        conn = connect_db()
        cur = conn.cursor()
        
        # Procesamos la inserción masiva y la expansión heurística
        total_afectados = process_expansion(cur, unique_seeds_with_bases)
        
        # Opcional: Acá podrías meter lógica para guardar el log en 'blocked_calls_log' si querés métricas de uso.
        conn.commit()
        
        return {
            "status": "success",
            "semillas_unicas_procesadas": len(unique_seeds_with_bases),
            "registros_bd_impactados": total_afectados
        }
    except Exception as e:
        if 'conn' in locals() and conn: conn.rollback()
        raise HTTPException(status_code=500, detail=f"Error interno del servidor: {str(e)}")
    finally:
        if 'conn' in locals() and conn:
            cur.close()
            conn.close()
